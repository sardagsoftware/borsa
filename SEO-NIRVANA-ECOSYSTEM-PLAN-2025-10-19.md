```json
  },

  }
}
```

### 2. FAQPage Schema (All Pages)

```json
{
  "@context": "https://schema.org",
  "@type": "FAQPage",
  "mainEntity": [
    {
      "@type": "Question",

      "acceptedAnswer": {
        "@type": "Answer",
```

```json
{
  "@context": "https://schema.org",
```
