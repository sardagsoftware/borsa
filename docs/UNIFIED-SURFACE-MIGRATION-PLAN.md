### State Management (Zustand)

```typescript
interface AppState {
  // Messages
  messages: Message[];
  addMessage: (msg: Message) => void;
  
  // Intents
  intents: Intent[];
  setIntents: (intents: Intent[]) => void;
  
  // Dock
  dock: {
    open: boolean;
    tab: 'overview' | 'health' | 'logs' | 'settings';
    vendor?: string;
  };
  openDock: (vendor: string, tab?: string) => void;
  closeDock: () => void;
  
  // Flags
  flags: FeatureFlags;
  setFlags: (flags: FeatureFlags) => void;
  
  // User
  user: {
    scopes: string[];
    locale: string;
    persona: string;
  };
  
  // UI State
  busy: boolean;
  setBusy: (busy: boolean) => void;
}
```
