   <h1 data-i18n="app.title">LyDian Medical AI</h1>
   <button data-i18n="buttons.submit">Submit</button>
   <input data-i18n-placeholder="chat.inputPlaceholder">

JavaScript API:
───────────────────────────────────────────────────────────────────────

// Get translation
const title = window.t('app.title');

// Change language
await window.medicalI18n.setLanguage('tr');

// Format numbers
const num = window.medicalI18n.formatNumber(1234.56);
// EN: "1,234.56", DE: "1.234,56", TR: "1.234,56"

// Format dates
const date = window.medicalI18n.formatDate(new Date());
// EN: "January 6, 2025", TR: "6 Ocak 2025"

// Check if RTL
if (window.medicalI18n.isRTL()) {
    console.log('RTL mode active');
}

// Listen for language changes
window.addEventListener('languageChanged', (event) => {
    console.log('Language:', event.detail.language);
    console.log('Is RTL:', event.detail.isRTL);
});
